@Injectable()
export class WorkspaceService {
  private readonly workspaceRoot = 'workspace/sessions';

  async initializeSession(sessionId: string): Promise<void> {
    const sessionPath = path.join(this.workspaceRoot, sessionId);

    // 创建目录结构
    await fs.ensureDir(path.join(sessionPath, 'code'));
    await fs.ensureDir(path.join(sessionPath, 'docs'));

    // 创建初始文件
    await fs.writeFile(
      path.join(sessionPath, 'metadata.json'),
      JSON.stringify({
        sessionId,
        createdAt: new Date(),
        version: '1.0.0'
      }, null, 2)
    );

    await fs.writeFile(
      path.join(sessionPath, 'transcripts.jsonl'),
      ''
    );

    await this.appendTranscript(sessionId, {
      type: 'session_created',
      sessionId,
      timestamp: new Date()
    });
  }

  async appendTranscript(sessionId: string, event: any): Promise<void> {
    const transcriptPath = path.join(
      this.workspaceRoot,
      sessionId,
      'transcripts.jsonl'
    );

    await fs.appendFile(
      transcriptPath,
      JSON.stringify({ ...event, timestamp: new Date() }) + '\n'
    );
  }
}

// src/agents/adapters/codex.adapter.ts
@Injectable()
export class CodexAdapter implements ILLMAdapter {
  readonly id = 'codex-001';
  readonly name = 'Codex';
  readonly type = 'codex';
  readonly role = '代码审查与质量把控';
  readonly callType = 'cli';

  async generate(prompt: string, context: AgentContext): Promise<AgentResponse> {
    const command = `codex-cli --prompt "${this.escapeShell(prompt)}" --format json`;
    
    const { stdout } = await execAsync(command, {
      timeout: 60000
    });

    return JSON.parse(stdout);
  }

  async shouldRespond(message: Message, context: AgentContext): Promise<Decision> {
    // 1. 被@提及
    if (message.content.includes('@Codex')) {
      return { should: true, reason: '被直接@提及' };
    }

    // 2. 关键词匹配
    const keywords = ['审查', '检查', '测试', '问题', 'bug'];
    if (keywords.some(k => message.content.includes(k))) {
      return { should: true, reason: '消息涉及代码审查' };
    }

    // 3. 检测到新代码
    if (context.workspaceChange?.type === 'file_added') {
      return { should: true, reason: '检测到新代码文件' };
    }

    return { should: false };
  }
}
```

// src/agents/services/decision-engine.service.ts
@Injectable()
export class AgentDecisionEngine {
  async decideResponders(
    message: Message,
    agents: ILLMAdapter[],
    context: AgentContext
  ): Promise<AgentDecision[]> {
    // 并行询问所有Agent
    const decisions = await Promise.allSettled(
      agents.map(async (agent) => {
        const decision = await agent.shouldRespond(message, context);
        return {
          agent,
          should: decision.should,
          reason: decision.reason,
          priority: this.calculatePriority(message, agent)
        };
      })
    );

    // 过滤出决定响应的Agent
    return decisions
      .filter(d => d.status === 'fulfilled' && d.value.should)
      .map(d => d.value)
      .sort((a, b) => b.priority - a.priority);
  }

  private calculatePriority(message: Message, agent: ILLMAdapter): number {
    // @提及 = 最高优先级
    if (message.content.includes(`@${agent.name}`)) {
      return 10;
    }

    // 关键词匹配 = 中等优先级
    // 其他 = 低优先级
    return 5;
  }
}
```

async *streamGenerate(
  prompt: string,
  context: AgentContext
): AsyncGenerator<string> {
  const response = await this.httpService.post(apiUrl, {
    model: this.model,
    messages: [{ role: 'user', content: prompt }],
    stream: true
  }, {
    responseType: 'stream'
  }).toPromise();

  for await (const chunk of response.data) {
    const lines = chunk.toString().split('\n');
    for (const line of lines) {
      if (line.startsWith('data: ')) {

        }
      }
    }
  }

@Injectable()
export class ShortTermMemoryService {
  constructor(@InjectRedis() private readonly redis: Redis) {}

  async save(sessionId: string, messages: Message[]): Promise<void> {
    const key = `memory:short:${sessionId}`;
    const recentMessages = messages.slice(-20);

    await this.redis.setex(
      key,
      300, // 5分钟
      JSON.stringify(recentMessages)
    );
  }

  async get(sessionId: string): Promise<Message[]> {
    const key = `memory:short:${sessionId}`;
    const data = await this.redis.get(key);
    return data ? JSON.parse(data) : [];
  }

  async append(sessionId: string, message: Message): Promise<void> {
    const messages = await this.get(sessionId);
    messages.push(message);
    await this.save(sessionId, messages);
  }
}

// src/config/config-watcher.service.ts
@Injectable()
export class ConfigWatcherService implements OnModuleInit {
  constructor(
    private readonly agentConfigService: AgentConfigService,
    private readonly eventEmitter: EventEmitter2,
    private readonly chatGateway: ChatGateway
  ) {}

  async onModuleInit() {
    const watcher = chokidar.watch([
      'config/agents.config.json',
      'config/prompts/**/*.md'
    ], {
      persistent: true,
      ignoreInitial: true,
      awaitWriteFinish: {
        stabilityThreshold: 500,
        pollInterval: 100
      }
    });

    watcher.on('change', async (filePath) => {
      console.log(`📝 配置文件变更: ${filePath}`);

      try {
        if (filePath.includes('agents.config.json')) {
          await this.reloadAgents();
        }
      } catch (error) {
        console.error('❌ 重载失败:', error);
      }
    });

    console.log('✅ Config Watcher已启动');
  }

  private async reloadAgents() {
    const startTime = Date.now();
    
    await this.agentConfigService.loadAgentsFromConfig();
    
    this.chatGateway.broadcast('system:notification', {

```typescript
// src/tasks/scheduled-tasks.service.ts
@Injectable()
export class ScheduledTasksService {
  @Cron(CronExpression.EVERY_5_MINUTES)
  async healthCheckAgents() {
    console.log('🏥 执行Agent健康检查...');
    
    const agents = await this.agentService.getAllAgents();
    
    for (const agent of agents) {
      const isHealthy = await agent.healthCheck();
      await this.agentService.updateStatus(
        agent.id,
        isHealthy ? 'online' : 'offline'
      );
    }
  }

  @Cron('0 2 * * *') // 每天凌晨2点
  async cleanupExpiredSessions() {
    console.log('🗑️ 清理过期会话...');
    
    const expiredDate = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);
    const expired = await this.sessionService.findExpired(expiredDate);
    
    for (const session of expired) {
      await this.workspaceService.archiveSession(session.id);
      await this.sessionService.markAsArchived(session.id);
    }
    
    console.log(`✅ 已清理 ${expired.length} 个过期会话`);
  }
}
```

### CI/CD流程

```yaml
# .github/workflows/ci.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - name: Install dependencies
        run: npm install
      - name: Run tests
        run: npm run test
      - name: Run linter
        run: npm run lint

  build:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - name: Build Docker image
        run: docker build -t lobster-backend .
      - name: Push to registry
        run: docker push lobster-backend:${{ github.sha }}

  deploy:
    needs: build
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    steps:
      - name: Deploy to production
        run: kubectl apply -f k8s/
```
